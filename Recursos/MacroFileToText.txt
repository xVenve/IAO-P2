Sub Archivos()

'Crea un archivo .txt por cada linea del Excel
'Llega hasta 400 filas; genera 400 documentos de texto en el directorio en el que se ejecute
    
    For i = 1 To 400 
        Data = Range("A" & i).Value
        Filename = i & ".txt" 'NOTA: Si no aparecen est√°n en Documentos. Se puede sustituir por: Filename = "ruta/del/directorio/seleccionado/" & i & ".txt"
        Open Filename For Append As #1
            Print #1, Data
        Close #1
    Next i


End Sub
